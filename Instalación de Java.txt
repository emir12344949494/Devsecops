1 Autenticasion como root 

sudo su 

2 Actualizacion de paquetes 

sudo apt update 

3 Instalacion de java 21 (jdk)

apt install -y openjdk-21-jdk

4 Configurar la variable de entorno para el uso de Java 

echo 'export JAVA_HOME=/usr/lib/jvm/java-21-openjdk-amd64' >> ~/.bashrc && source ~/.bashrc
