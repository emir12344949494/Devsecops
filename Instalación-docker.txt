1-Autenticasion como root 

sudo su

2 instalar dependencias

apt install -y apt-transport-https ca-certificates curl software-properties-common

3 Agregar la llave GPG oficial de Docker

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

4 Agregar el repositorio a las fuentes de APT

echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

5 Actualizar los paquetes
apt update 

6 Instalar docker bash 
apt install docker.io


7 Generar la imagen docker 
docker pull ubuntu

8 Corroborar la imagen descargada 

docker images 

9 Entrar al contenedor 
docker run -it --name p-ubuntu-prueba ubuntu /bin/bash

10 salir del comando 
exit
